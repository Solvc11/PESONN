<button onclick="calcular()">Calcular</button>

<div id="resultado" class="resultado"></div>

<script>

function calcular() {

    let pn = parseFloat(document.getElementById("pn").value);
    let d1 = parseFloat(document.getElementById("d1").value);
    let d2 = parseFloat(document.getElementById("d2").value);
    let d3 = parseFloat(document.getElementById("d3").value);

    if (!pn) {
        alert("Ingrese peso de nacimiento");
        return;
    }

    function calc(peso) {
        let perdida_g = pn - peso;
        let perdida_pct = (perdida_g / pn * 100);
        return {
            g: perdida_g,
            p: perdida_pct
        };
    }

    let r1 = d1 ? calc(d1) : null;
    let r2 = d2 ? calc(d2) : null;
    let r3 = d3 ? calc(d3) : null;

    let html = "";

    if (r1) {
        html += `<p class="${r1.p > 5.6 ? 'rojo':''}">
        Día 1: ↓ ${r1.g.toFixed(0)} g (${r1.p.toFixed(1)}%)
        </p>`;
    }

    if (r2) {
        html += `<p class="${r2.p > 10 ? 'rojo':''}">
        Día 2: ↓ ${r2.g.toFixed(0)} g (${r2.p.toFixed(1)}%)
        </p>`;
    }

    if (r3) {
        html += `<p class="${r3.p > 10 ? 'rojo':''}">
        Día 3: ↓ ${r3.g.toFixed(0)} g (${r3.p.toFixed(1)}%)
        </p>`;
    }

    document.getElementById("resultado").innerHTML = html;
}

</script>
